Simple List Table

Create a basic data table with columns and rows. This is the most common use case for displaying tabular data. 

// Sample data
const data = [
    { id: 1, name: 'John Doe', age: 30, city: 'New York', salary: 75000 },
    { id: 2, name: 'Jane Smith', age: 25, city: 'Los Angeles', salary: 80000 },
    { id: 3, name: 'Bob Johnson', age: 35, city: 'Chicago', salary: 70000 },
    { id: 4, name: 'Alice Brown', age: 28, city: 'Houston', salary: 85000 }
];

// Table configuration
const option = {
    container: document.getElementById('basic-list-table'),
    records: data,
    columns: [
        { field: 'id', title: 'ID', width: 80 },
        { field: 'name', title: 'Name', width: 150 },
        { field: 'age', title: 'Age', width: 80 },
        { field: 'city', title: 'City', width: 120 },
        { field: 'salary', title: 'Salary', width: 100 }
    ]
};

// Create the table
const table = new VTable.ListTable(option);

Basic Pivot Table

Create a pivot table to analyze and summarize your data with automatic grouping and aggregation. 

// Pivot table data
const pivotData = [
    { region: 'North', product: 'A', sales: 1000, quarter: 'Q1' },
    { region: 'North', product: 'B', sales: 1200, quarter: 'Q1' },
    { region: 'South', product: 'A', sales: 800, quarter: 'Q1' },
    { region: 'South', product: 'B', sales: 900, quarter: 'Q1' },
    { region: 'North', product: 'A', sales: 1100, quarter: 'Q2' },
    { region: 'North', product: 'B', sales: 1300, quarter: 'Q2' }
];

const pivotOption = {
    container: document.getElementById('basic-pivot-table'),
    records: pivotData,
    rows: ['region'],
    columns: ['quarter'],
    indicators: [
        { field: 'sales', title: 'Total Sales', aggregationType: VTable.TYPES.AggregationType.SUM }
    ]
};

const pivotTable = new VTable.PivotTable(pivotOption);

Conditional Styling

Apply different styles based on data values to highlight important information. 

const conditionalOption = {
    container: document.getElementById('conditional-table'),
    records: data,
    columns: [
        { field: 'id', title: 'ID', width: 80 },
        { field: 'name', title: 'Name', width: 150 },
        { field: 'age', title: 'Age', width: 80 },
        { field: 'city', title: 'City', width: 120 },
        { 
            field: 'salary', 
            title: 'Salary', 
            width: 100,
            style: {
                bgColor(args) {
                    const { row, col, dataValue } = args;
                    if (dataValue >= 80000) return '#d4edda'; // Green for high salary
                    if (dataValue >= 75000) return '#fff3cd'; // Yellow for medium salary
                    return '#f8d7da'; // Red for low salary
                },
                color(args) {
                    const { dataValue } = args;
                    if (dataValue >= 80000) return '#155724';
                    if (dataValue >= 75000) return '#856404';
                    return '#721c24';
                }
            }
        }
    ]
};

const conditionalTable = new VTable.ListTable(conditionalOption);

Progress Bars and Indicators

Add visual indicators like progress bars to make data more intuitive. 

const progressData = [
    { name: 'Project A', progress: 85, status: 'On Track' },

Event Handling

Add interactivity with comprehensive event handling for clicks, selections, and data changes. 

const eventsOption = {
    container: document.getElementById('events-table'),
    records: data,
    columns: [
        { field: 'id', title: 'ID', width: 80 },
        { field: 'name', title: 'Name', width: 150 },
        { field: 'age', title: 'Age', width: 80 },
        { field: 'city', title: 'City', width: 120 },
        { field: 'salary', title: 'Salary', width: 100 }
    ]
};

const eventsTable = new VTable.ListTable(eventsOption);
const logElement = document.getElementById('events-log');

function logEvent(message) {
    const timestamp = new Date().toLocaleTimeString();
    logElement.innerHTML += `[${timestamp}] ${message}\n`;
    logElement.scrollTop = logElement.scrollHeight;
}

// Cell click event
eventsTable.on('click_cell', (args) => {
    logEvent(`Cell clicked: row ${args.row}, col ${args.col}, value: ${args.value}`);
});

// Selection change
eventsTable.on('selected_cell', (args) => {
    logEvent(`Cell selected: row ${args.row}, col ${args.col}`);
});

// Mouse hover
eventsTable.on('mouseenter_cell', (args) => {
    logEvent(`Mouse entered cell: row ${args.row}, col ${args.col}`);
});
